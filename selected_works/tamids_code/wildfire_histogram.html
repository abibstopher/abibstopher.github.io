countdf = df.groupby(['Week/Year']).count()
count_dict = countdf.to_dict()['Timestamp']
week_count = {'2014': [0]*54,
              '2015': [0]*54,
              '2016': [0]*54,
              '2017': [0]*54,
              '2018': [0]*54,
              '2019': [0]*54,
              '2020': [0]*54,
              '2021': [0]*54,
              '2022': [0]*54}

for key, val in countdf.to_dict()['Timestamp'].items():
    year = key[-4:]
    if year in week_count:
        week = int(key[:-5])
        week_count[year][week - 1] = count_dict[key]
